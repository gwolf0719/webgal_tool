;========================================
; 檔案名稱: test.txt
; 描述: WebGAL 擴展功能測試場景
; 說明: 展示各種 WebGAL 命令和擴展功能
;========================================

;========================================
; 場景開始
;========================================

label:start;

changeBg:background.jpg -next;
bgm:music.mp3;

主角:歡迎使用 WebGAL Script Extension！;
主角:這個擴展提供了強大的編輯功能。 -v=voice1.ogg;

:現在讓我們來演示各種功能。;

setVar:score=0;
setVar:playerName=玩家;

;========================================
; 選擇分支演示
;========================================

changeFigure:character1.png -left -next;

主角:你想要了解哪個功能？;

choose:自動補全:feature_completion|語法診斷:feature_diagnostic|跳過:skip_demo;

;========================================
; 自動補全功能演示
;========================================

label:feature_completion;

主角:自動補全功能可以幫助你快速輸入命令。;
主角:只要輸入命令的前幾個字母，就會自動提示。;

setVar:score=score+10;

jumpLabel:continue;

;========================================
; 語法診斷功能演示
;========================================

label:feature_diagnostic;

主角:語法診斷功能會即時檢查你的腳本。;
主角:如果有錯誤，會立即顯示提示。;

setVar:score=score+20;

jumpLabel:continue;

;========================================
; 跳過演示
;========================================

label:skip_demo;

主角:那我們直接進入結局吧。;

;========================================
; 繼續劇情
;========================================

label:continue;

changeFigure:character2.png -right -next;
setAnimation:enter-from-right -target=fig-right -next;

主角:還有很多其他功能等你探索！;

miniAvatar:avatar.png;
主角:比如小頭像、動畫效果等等。 -v=voice2.ogg;

miniAvatar:none;

;========================================
; 結局判斷
;========================================

主角:根據你的選擇，會有不同的結局。;

jumpLabel:good_end -when=score>=15;
jumpLabel:normal_end;

;========================================
; 好結局
;========================================

label:good_end;

unlockCg:cg_good.jpg -name=完美結局;
unlockBgm:ending_theme.mp3 -name=結局主題曲;

主角:太棒了！你充分體驗了擴展的功能！;

callScene:ending_good.txt;

end;

;========================================
; 普通結局
;========================================

label:normal_end;

主角:希望這個擴展能幫助你更好地創作 WebGAL 作品！;

end;
